fun f(x, y) { return x + y; }
fun >>(x, y) { return x + y; }
//fun 1(x, y) { return x + y; }
print(f(1, 2), 1 >> 2);
print(4 * 2 >> 3);

fun fact(n) {
    let total = 1;
    while n > 1 {
        total = total * n;
        n = n - 1;
    }
    return total;
}

print(fact(3), fact(6), fact(8));
